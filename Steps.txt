1. Create folder
2. Open Terminal inside
3. Create Virtual Environment: python -m venv .venv
4. Activate Virtual Environment: git bash => source .venv/Scripts/activate
                            CMD/PowerShell => .venv\Scripts\activate
5. Install Django inside Virtual Environment: pip install django
6. Create New Project: django-admin startproject project .
6.5. Run Development Server to test project: python manage.py runserver
bind: localhost:8000 / 127.0.0.1:8000
to close server from terminal: ctrl + c

7. Create New Application: python manage.py startapp blog

8. Empty Database Created: Initialize Database with Django Default Tables
9. Generate SQL instructions in python: python manage.py makemigrations
10. Apply migrations to Database: python manage.py migrate

11. Create View Function in (blog/views.py): def func_name(request): HttpResponse(".......")
12. Create URLPattern in (project/urls.py): urlpatterns = [ path("route", func_name)]

# Step 1: Create Department
it = Department.objects.create(dep_name='IT')

# Step 2: Create Employee
ahmed = Employee.objects.create(
    emp_fn='Ahmed',
    emp_mn='Ali',
    emp_ln='Hassan',
    address='Cairo, Egypt',
    gender='Male',
    salary=3500.00,
    birthdate='1990-05-20',
    department=it
)

# Step 3: Create a second employee to act as supervisor
mona = Employee.objects.create(
    emp_fn='Mona',
    emp_ln='Youssef',
    address='Giza, Egypt',
    gender='Female',
    salary=4200.00,
    birthdate='1988-09-12',
    department=it,
    supervisor=ahmed
)

# Step 4: Assign manager to department
it.manager = ahmed
it.save()

# Step 5: Create a Project
erp = Project.objects.create(
    proj_name='ERP System',
    location='Smart Village',
    department=it
)

# Step 6: Link Employee to Project via WorksOn
WorksOn.objects.create(employee=ahmed, project=erp, hours_per_week=25)
WorksOn.objects.create(employee=mona, project=erp, hours_per_week=20)

# Step 7: Add Dependent
Dependent.objects.create(
    employee=ahmed,
    dep_name='Sara',
    gender='Female',
    birthdate='2015-05-01',
    relationship='Daughter'
)

# filter options
__exact: WHERE emp_fn = "Ahmed"
Employee.objects.filter(emp_fn__exact="Ahmed").delete()

__iexact
Employee.objects.filter(emp_fn__iexact="Ahmed").delete()

__contains: WHERE emp_fn LIKE "%a%"
Employee.objects.filter(emp_fn__contains="a").delete()

__icontains: WHERE emp_fn LIKE "%a%"
Employee.objects.filter(emp_fn__icontains="a").delete()

__startswith
Employee.objects.filter(emp_fn__startswith="a").delete()

__in: WHERE salary IN (...)
Employee.objects.filter(salary__in=[5000.00,6000.00]).delete()

__range: WHERE salary BETWEEN 1000.00 AND 10000.00
Employee.objects.filter(salary__range=(1000.00,10000.000)).delete()

__gt, __gte
Employee.objects.filter(salary__gte=1000.000).delete()

__lt, __lte
Employee.objects.filter(salary__lte=10000.000).delete()

def api(request): # Request Object
    print(request)
    print(type(request))
    print("Request Method:", request.method)
    print("Request Path:", request.path)
    print("Request GET:", request.GET)      # Path Parameters of GET (from URL) 
    print("Request POST:", request.POST)    # Path Parameters of POST (from Request Body) 
    print("Request body:", request.body)
    print("Request COOKIES:", request.COOKIES)
    print("Request FILES:", request.FILES)
    print("Request Headers:", request.META)
    print("Request get_host():", request.get_host())
    print("Request is_secure():", request.is_secure())
    # print("Request is_ajax():", request.is_ajax())
    # print("Request get_signed_cookie():", request.get_signed_cookie())
    return HttpResponse("Hello this is api endpoint")
